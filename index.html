And of course JS

For example


$( document ).ready(function() {
  var degree = 0;
  setInterval(function() {
    if (degree == 359) {
      degree = 0;
    } else {
      degree++; 
      $('.js-animation').css({'transform': 'rotate(' + degree + 'deg)'});
    }    								
  }, 20);
});
